for (Node n1: graph.getSelf().getNeighbors()) {
    for (Node n2: n1.getNeighbors()) {
        if(!n2.equals(graph.getSelf())) {
            graph.addEdge(graph.getSelf(), n2);
        }

                graph.addEdge(graph.getSelf(), n3);
            }
        }
    }
}
